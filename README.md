polkadot --chain=kusama

# Run Sidecar Service 
substrate-api-sidecar

# Command Line Use

main.py [-h] [-d DEPTH] [-e ERA] [-a] [-s SIDECAR_URL] account
positional arguments:
  account
optional arguments:
  -h, --help            show this help message and exit
  -d DEPTH, --depth DEPTH
  -e ERA, --era ERA
  -a, --all
  -s SIDECAR_URL, --sidecar-url SIDECAR_URL
